bool validateOwner(owner) {
  //TODO
  RegExp ownerRegex = new RegExp(r"^[a-zA-Z ]*$");
  if (owner == null) return false;
  try {
    return ownerRegex.hasMatch(owner);
  } catch (e) {
    return false;
  }
}
